### pages/admin/login/index.tsx - 管理员登录页

```typescript
import { useState } from 'react';
import { View, Text, Input, Button, Loading } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { adminLogin, isAdminLoggedIn } from '../../../utils/adminAuth';
import './index.css';

export default function AdminLogin() {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    // 检查是否已登录
    if (isAdminLoggedIn()) {
      Taro.redirectTo({
        url: '/pages/admin/dashboard/index',
      });
    }
  }, []);

  const handleLogin = async () => {
    if (!username || !password) {
      Taro.showToast({
        title: '请输入账号和密码',
        icon: 'none',
      });
      return;
    }

    setLoading(true);

    try {
      const adminInfo = await adminLogin(username, password);

      if (adminInfo) {
        Taro.showToast({
          title: '登录成功',
          icon: 'success',
        });

        // 跳转到仪表盘
        setTimeout(() => {
          Taro.redirectTo({
            url: '/pages/admin/dashboard/index',
          });
        }, 1000);
      } else {
        Taro.showToast({
          title: '账号或密码错误',
          icon: 'none',
        });
      }
    } catch (error) {
      console.error('[AdminLogin] 登录失败:', error);
      Taro.showToast({
        title: '登录失败，请重试',
        icon: 'none',
      });
    } finally {
      setLoading(false);
    }
  };

  return (
    <View className="admin-login-page">
      <View className="login-container">
        {/* Logo */}
        <View className="logo">
          <Text className="logo-icon">🏥</Text>
          <Text className="logo-text">健康管理后台</Text>
        </View>

        {/* 登录表单 */}
        <View className="login-form">
          <View className="form-item">
            <Text className="form-label">账号</Text>
            <Input
              className="form-input"
              type="text"
              placeholder="请输入管理员账号"
              value={username}
              onInput={(e) => setUsername(e.detail.value)}
            />
          </View>

          <View className="form-item">
            <Text className="form-label">密码</Text>
            <Input
              className="form-input"
              type="password"
              placeholder="请输入密码"
              value={password}
              onInput={(e) => setPassword(e.detail.value)}
            />
          </View>

          <Button 
            className="login-btn"
            onClick={handleLogin}
            disabled={loading}
          >
            {loading ? <Loading /> : '登录'}
          </Button>
        </View>

        {/* 版权信息 */}
        <View className="copyright">
          <Text className="copyright-text">
            © 2025 健康自检系统
          </Text>
        </View>
      </View>
    </View>
  );
}
```

### pages/admin/dashboard/index.tsx - 管理员仪表盘

```typescript
import { useState, useEffect } from 'react';
import { View, Text, ScrollView, Button, Loading } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { Statistics } from '../../../types';
import { getStatistics, getAllUsers } from '../../../utils/cloud';
import { adminLogout } from '../../../utils/adminAuth';
import LevelChart from './components/LevelChart';
import './index.css';

export default function AdminDashboard() {
  const [loading, setLoading] = useState(false);
  const [statistics, setStatistics] = useState<Statistics | null>(null);
  const [recentUsers, setRecentUsers] = useState<any[]>([]);

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    setLoading(true);

    try {
      // 获取统计数据
      const statsRes = await getStatistics();
      if (statsRes.success) {
        setStatistics(statsRes.data);
      }

      // 获取最近用户
      const usersRes = await getAllUsers({ limit: 5 });
      if (usersRes.success) {
        setRecentUsers(usersRes.data.users || []);
      }
    } catch (error) {
      console.error('[Dashboard] 加载数据失败:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleUserManagement = () => {
    Taro.navigateTo({
      url: '/pages/admin/user/index',
    });
  };

  const handleRecordManagement = () => {
    Taro.navigateTo({
      url: '/pages/admin/record/index',
    });
  };

  const handleExport = async () => {
    try {
      Taro.showLoading({ title: '导出中...' });
      
      const res = await Taro.cloud.callFunction({
        name: 'admin',
        data: {
          action: 'exportAllData',
        },
      });

      if (res.result.success) {
        Taro.showToast({
          title: '导出成功',
          icon: 'success',
        });
      } else {
        Taro.showToast({
          title: '导出失败',
          icon: 'none',
        });
      }
    } catch (error) {
      console.error('[Dashboard] 导出失败:', error);
    } finally {
      Taro.hideLoading();
    }
  };

  const handleLogout = () => {
    Taro.showModal({
      title: '提示',
      content: '确定要退出登录吗？',
      success: (res) => {
        if (res.confirm) {
          adminLogout();
          Taro.redirectTo({
            url: '/pages/admin/login/index',
          });
        }
      },
    });
  };

  return (
    <View className="admin-dashboard">
      {/* 头部 */}
      <View className="header">
        <Text className="header-title">数据概览</Text>
      </View>

      {/* 统计卡片 */}
      <View className="stats-container">
        <View className="stat-card">
          <Text className="stat-value">
            {loading ? '--' : statistics?.totalUsers || 0}
          </Text>
          <Text className="stat-label">总用户数</Text>
        </View>

        <View className="stat-card">
          <Text className="stat-value">
            {loading ? '--' : statistics?.totalChecks || 0}
          </Text>
          <Text className="stat-label">完成自检</Text>
        </View>

        <View className="stat-card">
          <Text className="stat-value">
            {loading ? '--' : statistics?.totalAnalysis || 0}
          </Text>
          <Text className="stat-label">完成分析</Text>
        </View>

        <View className="stat-card">
          <Text className="stat-value">
            {loading ? '--' : statistics?.totalPlans || 0}
          </Text>
          <Text className="stat-label">已选方案</Text>
        </View>
      </View>

      {/* 图表区域 */}
      {statistics && (
        <View className="chart-section">
          <Text className="chart-title">数据分布</Text>
          <LevelChart data={statistics} />
        </View>
      )}

      {/* 最近用户 */}
      <View className="recent-users">
        <Text className="section-title">最近用户</Text>
        <ScrollView className="user-list" scrollY>
          {recentUsers.map((user, index) => (
            <View key={user._id || index} className="user-item">
              <Text className="user-name">{user.nickName || '未命名'}</Text>
              <Text className="user-time">
                {new Date(user.createTime).toLocaleDateString()}
              </Text>
            </View>
          ))}
        </ScrollView>
      </View>

      {/* 快捷操作 */}
      <View className="quick-actions">
        <Button 
          className="action-btn"
          onClick={handleUserManagement}
        >
          用户管理
        </Button>
        <Button 
          className="action-btn"
          onClick={handleRecordManagement}
        >
          记录管理
        </Button>
        <Button 
          className="action-btn export-btn"
          onClick={handleExport}
        >
          导出数据
        </Button>
        <Button 
          className="action-btn logout-btn"
          onClick={handleLogout}
        >
          退出登录
        </Button>
      </View>
    </View>
  );
}
```

### pages/admin/dashboard/components/LevelChart.tsx - 等级图表组件

```typescript
import { View, Text } from '@tarojs/components';
import { Statistics } from '../../../../types';
import './index.css';

interface LevelChartProps {
  data: Statistics;
}

export default function LevelChart({ data }: LevelChartProps) {
  const items = [
    { label: '总用户', value: data.totalUsers, color: '#3B82F6' },
    { label: '自检', value: data.totalChecks, color: '#10B981' },
    { label: '分析', value: data.totalAnalysis, color: '#F59E0B' },
    { label: '方案', value: data.totalPlans, color: '#EF4444' },
  ];

  const maxValue = Math.max(...items.map(item => item.value));

  return (
    <View className="level-chart">
      {items.map((item, index) => (
        <View key={index} className="chart-item">
          <Text className="chart-label">{item.label}</Text>
          <View className="chart-bar">
            <View 
              className="bar-fill"
              style={{
                width: `${(item.value / maxValue) * 100}%`,
                backgroundColor: item.color,
              }}
            />
          </View>
          <Text className="chart-value">{item.value}</Text>
        </View>
      ))}
    </View>
  );
}
```

### pages/admin/user/index.tsx - 用户管理页

```typescript
import { useState, useEffect } from 'react';
import { View, Text, Input, ScrollView, Button, Loading } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { getAllUsers } from '../../../utils/cloud';
import './index.css';

export default function AdminUser() {
  const [loading, setLoading] = useState(false);
  const [users, setUsers] = useState<any[]>([]);
  const [keyword, setKeyword] = useState('');
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);

  useEffect(() => {
    loadUsers();
  }, [page, keyword]);

  const loadUsers = async () => {
    setLoading(true);

    try {
      const res = await getAllUsers({
        skip: (page - 1) * 10,
        limit: 10,
        search: keyword,
      });

      if (res.success) {
        const newUsers = res.data.users || [];
        setHasMore(newUsers.length === 10);

        if (page === 1) {
          setUsers(newUsers);
        } else {
          setUsers([...users, ...newUsers]);
        }
      }
    } catch (error) {
      console.error('[User] 加载用户失败:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleSearch = () => {
    setPage(1);
    loadUsers();
  };

  const handleLoadMore = () => {
    if (!hasMore || loading) return;
    setPage(page + 1);
  };

  const handleViewDetail = (userId: string) => {
    Taro.navigateTo({
      url: `/pages/admin/user/detail?id=${userId}`,
    });
  };

  const handleExport = async () => {
    try {
      Taro.showLoading({ title: '导出中...' });
      
      const res = await Taro.cloud.callFunction({
        name: 'admin',
        data: {
          action: 'exportUsers',
          keyword,
        },
      });

      if (res.result.success) {
        Taro.showToast({
          title: '导出成功',
          icon: 'success',
        });
      }
    } catch (error) {
      console.error('[User] 导出失败:', error);
    } finally {
      Taro.hideLoading();
    }
  };

  return (
    <View className="admin-user">
      {/* 搜索栏 */}
      <View className="search-bar">
        <Input
          className="search-input"
          placeholder="搜索用户姓名或手机号"
          value={keyword}
          onInput={(e) => setKeyword(e.detail.value)}
        />
        <Button className="search-btn" onClick={handleSearch}>
          搜索
        </Button>
      </View>

      {/* 用户列表 */}
      <ScrollView 
        className="user-list"
        scrollY
        onScrollToLower={handleLoadMore}
      >
        {users.map((user, index) => (
          <View key={user._id || index} className="user-item">
            <View className="user-info">
              <Text className="user-name">{user.nickName || '未命名'}</Text>
              <Text className="user-phone">{user.phone || '暂无手机号'}</Text>
            </View>
            <View className="user-stats">
              <Text className="stat">
                自检: {user.checkCount || 0}次
              </Text>
            </View>
            <Button 
              className="detail-btn"
              size="mini"
              onClick={() => handleViewDetail(user._id)}
            >
              详情
            </Button>
          </View>
        ))}

        {loading && (
          <View className="loading">
            <Loading />
          </View>
        )}

        {!hasMore && users.length > 0 && (
          <Text className="no-more">没有更多了</Text>
        )}
      </ScrollView>

      {/* 底部操作 */}
      <View className="footer">
        <Button className="export-btn" onClick={handleExport}>
          导出用户数据
        </Button>
      </View>
    </View>
  );
}
```

### pages/admin/record/index.tsx - 记录管理页

```typescript
import { useState, useEffect } from 'react';
import { View, Text, Input, ScrollView, Button, Loading, Picker } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { getAllUsers } from '../../../utils/cloud';
import './index.css';

const RECORD_TYPES = [
  { label: '全部', value: '' },
  { label: '症状检查', value: 'symptom' },
  { label: '健康分析', value: 'analysis' },
  { label: '方案选择', value: 'plan' },
];

export default function AdminRecord() {
  const [loading, setLoading] = useState(false);
  const [records, setRecords] = useState<any[]>([]);
  const [keyword, setKeyword] = useState('');
  const [recordType, setRecordType] = useState(0);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);

  useEffect(() => {
    loadRecords();
  }, [page, recordType]);

  const loadRecords = async () => {
    setLoading(true);

    try {
      const res = await Taro.cloud.callFunction({
        name: 'admin',
        data: {
          action: 'getRecords',
          type: RECORD_TYPES[recordType].value,
          keyword,
          skip: (page - 1) * 10,
          limit: 10,
        },
      });

      if (res.result.success) {
        const newRecords = res.result.data.records || [];
        setHasMore(newRecords.length === 10);

        if (page === 1) {
          setRecords(newRecords);
        } else {
          setRecords([...records, ...newRecords]);
        }
      }
    } catch (error) {
      console.error('[Record] 加载记录失败:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleSearch = () => {
    setPage(1);
    loadRecords();
  };

  const handleLoadMore = () => {
    if (!hasMore || loading) return;
    setPage(page + 1);
  };

  const handleTypeChange = (e) => {
    setRecordType(e.detail.value);
    setPage(1);
  };

  const handleExport = async () => {
    try {
      Taro.showLoading({ title: '导出中...' });
      
      const res = await Taro.cloud.callFunction({
        name: 'admin',
        data: {
          action: 'exportRecords',
          type: RECORD_TYPES[recordType].value,
          keyword,
        },
      });

      if (res.result.success) {
        Taro.showToast({
          title: '导出成功',
          icon: 'success',
        });
      }
    } catch (error) {
      console.error('[Record] 导出失败:', error);
    } finally {
      Taro.hideLoading();
    }
  };

  return (
    <View className="admin-record">
      {/* 筛选栏 */}
      <View className="filter-bar">
        <Picker
          className="type-picker"
          mode="selector"
          range={RECORD_TYPES.map(item => item.label)}
          value={recordType}
          onChange={handleTypeChange}
        >
          <Text className="picker-text">
            {RECORD_TYPES[recordType].label}
          </Text>
        </Picker>
        <Input
          className="search-input"
          placeholder="搜索用户姓名或手机号"
          value={keyword}
          onInput={(e) => setKeyword(e.detail.value)}
        />
        <Button className="search-btn" onClick={handleSearch}>
          搜索
        </Button>
      </View>

      {/* 记录列表 */}
      <ScrollView 
        className="record-list"
        scrollY
        onScrollToLower={handleLoadMore}
      >
        {records.map((record, index) => (
          <View key={record._id || index} className="record-item">
            <View className="record-header">
              <Text className="record-user">{record.userName || '未命名'}</Text>
              <Text className="record-type">{record.type}</Text>
            </View>
            <View className="record-detail">
              <Text className="record-time">
                {new Date(record.createTime).toLocaleString()}
              </Text>
            </View>
          </View>
        ))}

        {loading && (
          <View className="loading">
            <Loading />
          </View>
        )}

        {!hasMore && records.length > 0 && (
          <Text className="no-more">没有更多了</Text>
        )}
      </ScrollView>

      {/* 底部操作 */}
      <View className="footer">
        <Button className="export-btn" onClick={handleExport}>
          导出记录数据
        </Button>
      </View>
    </View>
  );
}
```
