```typescript
const handleViewHistory = async (user: any) => {
  setLoadingHistory(true);
  setHistoryPhone(user.phone || user.name);
  try {
    // 优先使用 phoneGroupId 查询，其次使用 phone，最后使用 name
    let queryParam = '';
    if (user.phoneGroupId) {
      queryParam = `phoneGroupId=${encodeURIComponent(user.phoneGroupId)}`;
    } else if (user.phone) {
      queryParam = `phone=${encodeURIComponent(user.phone)}`;
    } else {
      queryParam = `name=${encodeURIComponent(user.name)}`;
    }

    const response = await fetch(`/api/user/history?${queryParam}`);
    const data = await response.json();
    if (data.success) {
      setHistoryUsers(data.users);
      setShowHistoryDialog(true);
    } else {
      alert('获取历史记录失败');
    }
  } catch (error) {
    console.error('Failed to fetch user history:', error);
    alert('获取历史记录失败，请重试');
  } finally {
    setLoadingHistory(false);
  }
};
```

### 文件2：src/lib/formDataManager.ts

**新增**：服务器端安全的表单数据管理器

```typescript
'use client';

/**
 * 表单数据管理器
 * 提供服务器端安全的 localStorage 操作
 */

export class FormDataManager {
  /**
   * 安全地从 localStorage 读取数据
   * @param key 存储键
   * @returns 数据或 null
   */
  static getSafe(key: string): any | null {
    // 检查是否在浏览器环境
    if (typeof window === 'undefined') {
      return null;
    }

    try {
      const data = localStorage.getItem(key);
      return data ? JSON.parse(data) : null;
    } catch (error) {
      console.error(`Error reading from localStorage (${key}):`, error);
      return null;
    }
  }

  /**
   * 安全地写入数据到 localStorage
   * @param key 存储键
   * @param value 要存储的数据
   * @returns 是否成功
   */
  static setSafe(key: string, value: any): boolean {
    // 检查是否在浏览器环境
    if (typeof window === 'undefined') {
      return false;
    }

    try {
      localStorage.setItem(key, JSON.stringify(value));
      return true;
    } catch (error) {
      console.error(`Error writing to localStorage (${key}):`, error);
      return false;
    }
  }

  /**
   * 安全地从 localStorage 删除数据
   * @param key 存储键
   * @returns 是否成功
   */
  static removeSafe(key: string): boolean {
    // 检查是否在浏览器环境
    if (typeof window === 'undefined') {
      return false;
    }

    try {
      localStorage.removeItem(key);
      return true;
    } catch (error) {
      console.error(`Error removing from localStorage (${key}):`, error);
      return false;
    }
  }

  /**
   * 清空所有表单相关数据
   */
  static clearAll(): void {
    if (typeof window !== 'undefined') {
      const keys = [
        'selectedSymptoms',
        'selectedBadHabitsRequirements',
        'selectedSymptoms300',
        'healthAnalysis',
        'sevenQuestionsData',
        'userInfo',
      ];
      keys.forEach(key => this.removeSafe(key));
    }
  }
}

export default FormDataManager;
```
