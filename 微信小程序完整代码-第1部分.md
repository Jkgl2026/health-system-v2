```typescript
import Taro from '@tarojs/taro';

const STORAGE_PREFIX = 'health_';

/**
 * 设置本地存储
 */
export const setStorage = (key: string, value: any): void => {
  try {
    Taro.setStorageSync(STORAGE_PREFIX + key, value);
    console.log(`[Storage] 设置 ${key} 成功`);
  } catch (error) {
    console.error(`[Storage] 设置 ${key} 失败:`, error);
  }
};

/**
 * 获取本地存储
 */
export const getStorage = <T = any>(key: string): T | null => {
  try {
    const value = Taro.getStorageSync(STORAGE_PREFIX + key);
    return value || null;
  } catch (error) {
    console.error(`[Storage] 获取 ${key} 失败:`, error);
    return null;
  }
};

/**
 * 删除本地存储
 */
export const removeStorage = (key: string): void => {
  try {
    Taro.removeStorageSync(STORAGE_PREFIX + key);
    console.log(`[Storage] 删除 ${key} 成功`);
  } catch (error) {
    console.error(`[Storage] 删除 ${key} 失败:`, error);

};

/**
 * 清空所有本地存储
 */

    console.log('[Storage] 清空所有存储成功');

    console.error('[Storage] 清空存储失败:', error);

};
```

### utils/adminAuth.ts - 管理员认证工具

```typescript
import Taro from '@tarojs/taro';
import { AdminInfo } from '../types';

const ADMIN_TOKEN_KEY = 'admin_token';
const ADMIN_INFO_KEY = 'admin_info';

/**
 * 管理员登录
 */
export const adminLogin = async (username: string, password: string): Promise<AdminInfo | null> => {
  try {
    const res = await Taro.cloud.callFunction({
      name: 'admin',
      data: {
        action: 'login',
        username,
        password,
      },
    });

    if (res.result.success) {
      const adminInfo: AdminInfo = res.result.data;
      
      // 存储Token和用户信息
      Taro.setStorageSync(ADMIN_TOKEN_KEY, res.result.token);
      Taro.setStorageSync(ADMIN_INFO_KEY, adminInfo);
      
      console.log('[AdminAuth] 登录成功:', adminInfo.username);
      return adminInfo;
    } else {
      console.error('[AdminAuth] 登录失败:', res.result.error);
      return null;
    }
  } catch (error) {
    console.error('[AdminAuth] 登录异常:', error);
    return null;
  }
};

/**
 * 管理员登出
 */
export const adminLogout = (): void => {
  try {
    Taro.removeStorageSync(ADMIN_TOKEN_KEY);
    Taro.removeStorageSync(ADMIN_INFO_KEY);
    console.log('[AdminAuth] 登出成功');
  } catch (error) {
    console.error('[AdminAuth] 登出失败:', error);
  }
};

/**
 * 获取管理员Token
 */
export const getAdminToken = (): string | null => {
  try {
    return Taro.getStorageSync(ADMIN_TOKEN_KEY);
  } catch (error) {
    return null;
  }
};

/**
 * 获取管理员信息
 */
export const getAdminInfo = (): AdminInfo | null => {
  try {
    return Taro.getStorageSync(ADMIN_INFO_KEY);
  } catch (error) {
    return null;
  }
};

/**
 * 检查管理员是否已登录
 */
export const isAdminLoggedIn = (): boolean => {
  const token = getAdminToken();
  return !!token;
};

/**
 * 验证管理员Token
 */
export const verifyAdminToken = async (): Promise<boolean> => {
  try {
    const token = getAdminToken();
    if (!token) return false;

    const res = await Taro.cloud.callFunction({
      name: 'admin',
      data: {
        action: 'verify',
        token,
      },
    });

    return res.result.success;
  } catch (error) {
    console.error('[AdminAuth] Token验证失败:', error);
    return false;
  }
};
```

```typescript
import Taro from '@tarojs/taro';

/**
 * 调用云函数
 */
export const callCloudFunction = async <T = any>(
  name: string,
  data: any = {}
): Promise<ApiResponse<T>> => {
  try {
    const res = await Taro.cloud.callFunction({
      name,
      data,
    });

    console.log(`[CloudFunction] 调用 ${name} 成功:`, res.result);
    return res.result as ApiResponse<T>;
  } catch (error) {
    console.error(`[CloudFunction] 调用 ${name} 失败:`, error);
    return {
      success: false,
      error: '调用云函数失败',
    };
  }
};

/**
 * 获取用户记录

export const getUserRecords = async (userId: string): Promise<ApiResponse> => {
```

```typescript
import { HealthAnalysis } from '../types';

/**
 * 计算综合健康评分
 */
export const calculateComprehensiveHealthScore = (
  symptomScore: number,
  healthAnalysis: Partial<HealthAnalysis>
): HealthAnalysis => {
  // 基于症状评分调整健康指标
  const symptomFactor = 1 - (symptomScore / 100);
  
  const baseScores = {
    qiAndBlood: 70,
    circulation: 65,
    toxins: 60,
    bloodLipids: 70,
    coldness: 55,
    immunity: 65,
    emotions: 70,
    overallHealth: 65,
  };

  // 应用症状影响
  const analysis: HealthAnalysis = {
    qiAndBlood: Math.round(baseScores.qiAndBlood * symptomFactor),
    circulation: Math.round(baseScores.circulation * symptomFactor),
    toxins: Math.round(baseScores.toxins * symptomFactor),
    bloodLipids: Math.round(baseScores.bloodLipids * symptomFactor),
    coldness: Math.round(baseScores.coldness * symptomFactor),
    immunity: Math.round(baseScores.immunity * symptomFactor),
    emotions: Math.round(baseScores.emotions * symptomFactor),
    overallHealth: Math.round(baseScores.overallHealth * symptomFactor),
  };

  return analysis;
};

/**
 * 计算BMI
 */
export const calculateBMI = (height: number, weight: number): number => {
  if (!height || !weight) return 0;
  const heightInMeters = height / 100;
  return Number((weight / (heightInMeters * heightInMeters)).toFixed(1));
};

/**
 * 获取BMI等级
 */
export const getBMILevel = (bmi: number): string => {
  if (bmi < 18.5) return '体重过轻';
  if (bmi < 24) return '体重正常';
  if (bmi < 28) return '超重';
  return '肥胖';
};

/**
 * 获取BMI建议
 */
export const getBMISuggestion = (bmi: number): string => {
  const level = getBMILevel(bmi);
  
  switch (level) {
    case '体重过轻':
      return '建议增加营养摄入，适当增重，加强肌肉锻炼。';

    case '超重':
```
