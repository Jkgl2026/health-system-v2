```typescript
'use client';

import { useState, useEffect } from 'react';
import { 
  View, 
  Text, 
  Input, 
  Button, 
  Loading, 
  ScrollView 
} from '@tarojs/components';
import Taro from '@tarojs/taro';
import './index.css';

interface UserInfo {
  _id: string;
  name: string;
  phone: string;
  gender: string;
  age: number;
  height: string;
  weight: string;
  occupation: string;
  bmi?: number;
  bmiLevel?: string;
  systolicBP?: number;
  diastolicBP?: number;
  bpLevel?: string;
}

interface HealthAnalysis {
  _id: string;
  userId: string;
  qiAndBlood: number;
  circulation: number;
  toxins: number;
  bloodLipids: number;
  coldness: number;
  immunity: number;
  emotions: number;
  overallHealth: number;
}

interface SymptomCheck {
  _id: string;
  userId: string;
  symptoms: number[];
  totalScore: number;
  checkedAt: string;
}

interface HealthPlan {
  _id: string;
  userId: string;
  planType: string;
  planDescription: string;
}

interface RequirementStatus {
  _id: string;
  userId: string;
  requirement1Completed: boolean;
  requirement2Completed: boolean;
  requirement3Completed: boolean;
  requirement4Completed: boolean;
}

export default function Compare() {
  const [loading, setLoading] = useState(false);
  const [searchKeyword, setSearchKeyword] = useState('');
  const [user1, setUser1] = useState<UserInfo | null>(null);
  const [user2, setUser2] = useState<UserInfo | null>(null);
  const [history1, setHistory1] = useState<any[]>([]);
  const [history2, setHistory2] = useState<any[]>([]);
  const [showResult, setShowResult] = useState(false);

  // 模拟用户历史记录数据
  const mockFetchUserHistory = async (userId: string) => {
    return new Promise((resolve) => {
      setTimeout(() => {
        const mockData = [
          {
            id: '1',
            userId,
            date: '2025-01-20',
            bmi: 22.5,
            bmiLevel: '正常',
            systolicBP: 120,
            diastolicBP: 80,
            bpLevel: '正常',
            healthAnalysis: {
              qiAndBlood: 75,
              circulation: 70,
              toxins: 65,
              bloodLipids: 72,
              coldness: 68,
              immunity: 71,
              emotions: 74,
              overallHealth: 70,
            },
            symptoms: [101, 201, 301],
            symptomScore: 15,
            plan: '调理方案A',
            requirements: {
              requirement1Completed: true,
              requirement2Completed: false,
              requirement3Completed: true,
              requirement4Completed: false,
            },
          },
          {
            id: '2',
            userId,
            date: '2025-01-25',
            bmi: 23.0,
            bmiLevel: '正常',
            systolicBP: 118,
            diastolicBP: 78,
            bpLevel: '正常',
            healthAnalysis: {
              qiAndBlood: 78,
              circulation: 73,
              toxins: 68,
              bloodLipids: 70,
              coldness: 66,
              immunity: 74,
              emotions: 76,
              overallHealth: 73,
            },
            symptoms: [102, 202],
            symptomScore: 10,
            plan: '调理方案B',
            requirements: {
              requirement1Completed: true,
              requirement2Completed: true,
              requirement3Completed: true,
              requirement4Completed: false,
            },
          },
        ];
        resolve(mockData);
      }, 500);
    });
  };

  const mockFetchUserFullData = async (userId: string) => {
    return new Promise((resolve) => {
      setTimeout(() => {
        const mockUser: UserInfo = {
          _id: userId,
          name: userId === 'user1' ? '李四' : '王五',
          phone: userId === 'user1' ? '13800138001' : '13800138002',
          gender: userId === 'user1' ? '男' : '女',
          age: userId === 'user1' ? 29 : 28,
          height: '165',
          weight: '70',
          occupation: '软件工程师',
          bmi: 22.5,
          bmiLevel: '正常',
          systolicBP: 120,
          diastolicBP: 80,
          bpLevel: '正常',
        };
        resolve(mockUser);
      }, 300);
    });
  };

  const handleSearch = async () => {
    if (!searchKeyword) {
      Taro.showToast({
        title: '请输入用户ID',
        icon: 'none',
      });
      return;
    }

    setLoading(true);
    try {
      const userData = await mockFetchUserFullData(searchKeyword);
      
      if (!user1) {
        setUser1(userData as UserInfo);
        const history = await mockFetchUserHistory(searchKeyword);
        setHistory1(history as any[]);
        Taro.showToast({
          title: '已选择第一个用户',
          icon: 'success',
        });
      } else if (!user2) {
        setUser2(userData as UserInfo);
        const history = await mockFetchUserHistory(searchKeyword);
        setHistory2(history as any[]);
        Taro.showToast({
          title: '已选择第二个用户',
          icon: 'success',
        });
      } else {
        Taro.showToast({
          title: '已选择两个用户，请点击对比',
          icon: 'none',
        });
      }
    } catch (error) {
      console.error('[Compare] 搜索用户失败:', error);
      Taro.showToast({
        title: '搜索失败，请重试',
        icon: 'none',
      });
    } finally {
      setLoading(false);
    }
  };

  const handleCompare = () => {
    if (!user1 || !user2) {
      Taro.showToast({
        title: '请先选择两个用户',
        icon: 'none',
      });
      return;
    }
    setShowResult(true);
  };

  const handleReset = () => {
    setUser1(null);
    setUser2(null);
    setHistory1([]);
    setHistory2([]);
    setShowResult(false);
    setSearchKeyword('');
  };

  const getBMIColor = (bmi?: number) => {
    if (!bmi) return 'text-gray-500';
    if (bmi < 18.5) return 'text-blue-500';
    if (bmi < 24) return 'text-green-500';
    if (bmi < 28) return 'text-yellow-500';
    return 'text-red-500';
  };

  const getBPColor = (systolicBP?: number) => {
    if (!systolicBP) return 'text-gray-500';
    if (systolicBP < 120) return 'text-green-500';
    if (systolicBP < 140) return 'text-yellow-500';
    return 'text-red-500';
  };

  const getHealthScoreColor = (score: number) => {
    if (score >= 80) return 'text-green-500';
    if (score >= 60) return 'text-yellow-500';
    return 'text-red-500';
  };

  const compareValue = (v1: any, v2: any, isBetter: 'higher' | 'lower' | 'normal' = 'normal') => {
    if (v1 === undefined || v2 === undefined) return { diff: 0, icon: '', color: 'text-gray-500' };
    
    const diff = v1 - v2;
    if (diff === 0) return { diff: 0, icon: '=', color: 'text-gray-500' };
    
    if (isBetter === 'higher') {
      return {
        diff,
        icon: diff > 0 ? '↑' : '↓',
        color: diff > 0 ? 'text-green-500' : 'text-red-500',
      };
    } else if (isBetter === 'lower') {
      return {
        diff,
        icon: diff < 0 ? '↑' : '↓',
        color: diff < 0 ? 'text-green-500' : 'text-red-500',
      };
    } else {
      return {
        diff,
        icon: diff > 0 ? '↑' : '↓',
        color: 'text-blue-500',
      };
    }
  };

  return (
    <View className="compare-page">
      {/* 搜索区域 */}
      <View className="search-section">
        <Text className="section-title">数据对比</Text>
        <View className="search-bar">
          <Input
            className="search-input"
            placeholder="输入用户ID（如：user1）"
            value={searchKeyword}
            onInput={(e) => setSearchKeyword(e.detail.value)}
          />
          <Button className="search-btn" onClick={handleSearch} disabled={loading}>
            {loading ? <Loading /> : '搜索'}
          </Button>
        </View>

        {/* 已选用户 */}
        <View className="selected-users">
          <View className={`user-card ${user1 ? 'selected' : ''}`}>
            <Text className="user-label">用户1</Text>
            {user1 ? (
              <>
                <Text className="user-name">{user1.name}</Text>
                <Text className="user-phone">{user1.phone}</Text>
              </>
            ) : (
              <Text className="user-placeholder">未选择</Text>
            )}
          </View>

          <Text className="vs-text">VS</Text>

          <View className={`user-card ${user2 ? 'selected' : ''}`}>
            <Text className="user-label">用户2</Text>
            {user2 ? (
              <>
                <Text className="user-name">{user2.name}</Text>
                <Text className="user-phone">{user2.phone}</Text>
              </>
            ) : (
              <Text className="user-placeholder">未选择</Text>
            )}
          </View>
        </View>

        {/* 操作按钮 */}
        <View className="action-buttons">
          <Button className="compare-btn" onClick={handleCompare} disabled={!user1 || !user2}>
            开始对比
          </Button>
          <Button className="reset-btn" onClick={handleReset}>
            重置
          </Button>
        </View>
      </View>

      {/* 对比结果 */}
      {showResult && user1 && user2 && (
        <ScrollView className="compare-result" scrollY>
          {/* 基本信息对比 */}
          <View className="compare-section">
            <Text className="section-title">基本信息对比</Text>
            
            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">姓名</Text>
                <Text className="item-value">{user1.name}</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">姓名</Text>
                <Text className="item-value">{user2.name}</Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">性别</Text>
                <Text className="item-value">{user1.gender}</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">性别</Text>
                <Text className="item-value">{user2.gender}</Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">年龄</Text>
                <Text className="item-value">{user1.age}岁</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">年龄</Text>
                <Text className="item-value">{user2.age}岁</Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">身高</Text>
                <Text className="item-value">{user1.height}cm</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">身高</Text>
                <Text className="item-value">{user2.height}cm</Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">体重</Text>
                <Text className="item-value">{user1.weight}kg</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">体重</Text>
                <Text className="item-value">{user2.weight}kg</Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">职业</Text>
                <Text className="item-value">{user1.occupation || '暂无'}</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">职业</Text>
                <Text className="item-value">{user2.occupation || '暂无'}</Text>
              </View>
            </View>
          </View>

          {/* BMI对比 */}
          <View className="compare-section">
            <Text className="section-title">BMI对比</Text>
            
            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">BMI值</Text>
                <Text className={`item-value ${getBMIColor(user1.bmi)}`}>
                  {user1.bmi?.toFixed(1) || '-'}
                </Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">BMI值</Text>
                <Text className={`item-value ${getBMIColor(user2.bmi)}`}>
                  {user2.bmi?.toFixed(1) || '-'}
                </Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">BMI等级</Text>
                <Text className="item-value">{user1.bmiLevel || '-'}</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">BMI等级</Text>
                <Text className="item-value">{user2.bmiLevel || '-'}</Text>
              </View>
            </View>
          </View>

          {/* 血压对比 */}
          <View className="compare-section">
            <Text className="section-title">血压对比</Text>
            
            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">收缩压</Text>
                <Text className={`item-value ${getBPColor(user1.systolicBP)}`}>
                  {user1.systolicBP || '-'}
                </Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">收缩压</Text>
                <Text className={`item-value ${getBPColor(user2.systolicBP)}`}>
                  {user2.systolicBP || '-'}
                </Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">舒张压</Text>
                <Text className={`item-value ${getBPColor(user1.diastolicBP)}`}>
                  {user1.diastolicBP || '-'}
                </Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">舒张压</Text>
                <Text className={`item-value ${getBPColor(user2.diastolicBP)}`}>
                  {user2.diastolicBP || '-'}
                </Text>
              </View>
            </View>

            <View className="compare-row">
              <View className="compare-item">
                <Text className="item-label">血压等级</Text>
                <Text className="item-value">{user1.bpLevel || '-'}</Text>
              </View>
              <View className="compare-item">
                <Text className="item-label">血压等级</Text>
                <Text className="item-value">{user2.bpLevel || '-'}</Text>
              </View>
            </View>
          </View>

          {/* 健康要素对比 */}
          {history1.length > 0 && history2.length > 0 && (
            <View className="compare-section">
              <Text className="section-title">健康要素对比</Text>
              
              {['qiAndBlood', 'circulation', 'toxins', 'bloodLipids', 'coldness', 'immunity', 'emotions', 'overallHealth'].map((key) => {
                const labels: any = {
                  qiAndBlood: '气血',
                  circulation: '循环',
                  toxins: '毒素',
                  bloodLipids: '血脂',
                  coldness: '寒凉',
                  immunity: '免疫力',
                  emotions: '情绪',
                  overallHealth: '整体健康',
                };
                
                const v1 = history1[0]?.healthAnalysis?.[key as keyof HealthAnalysis];
                const v2 = history2[0]?.healthAnalysis?.[key as keyof HealthAnalysis];
                const comparison = compareValue(v1, v2, 'higher');

                return (
                  <View key={key} className="compare-row-highlight">
                    <View className="compare-item">
                      <Text className="item-label">{labels[key]}</Text>
                      <Text className={`item-value ${getHealthScoreColor(v1)}`}>
                        {v1 || '-'}分
                      </Text>
                    </View>
                    <View className="compare-diff">
                      <Text className={`diff-icon ${comparison.color}`}>
                        {comparison.icon}
                      </Text>
                    </View>
                    <View className="compare-item">
                      <Text className="item-label">{labels[key]}</Text>
                      <Text className={`item-value ${getHealthScoreColor(v2)}`}>
                        {v2 || '-'}分
                      </Text>
                    </View>
                  </View>
                );
              })}
            </View>
          )}

          {/* 症状检查对比 */}
          {history1.length > 0 && history2.length > 0 && (
            <View className="compare-section">
              <Text className="section-title">症状检查对比</Text>
              
              <View className="compare-row-highlight">
                <View className="compare-item">
                  <Text className="item-label">症状得分</Text>
                  <Text className="item-value">
                    {history1[0]?.symptomScore || '-'}分
                  </Text>
                </View>
                <View className="compare-item">
                  <Text className="item-label">症状得分</Text>
                  <Text className="item-value">
                    {history2[0]?.symptomScore || '-'}分
                  </Text>
                </View>
              </View>

              <View className="compare-row">
                <View className="compare-item">
                  <Text className="item-label">症状数量</Text>
                  <Text className="item-value">
                    {history1[0]?.symptoms?.length || 0}个
                  </Text>
                </View>
                <View className="compare-item">
                  <Text className="item-label">症状数量</Text>
                  <Text className="item-value">
                    {history2[0]?.symptoms?.length || 0}个
                  </Text>
                </View>
              </View>
            </View>
          )}

          {/* 健康方案对比 */}
          {history1.length > 0 && history2.length > 0 && (
            <View className="compare-section">
              <Text className="section-title">健康方案对比</Text>
              
              <View className="compare-row">
                <View className="compare-item">
                  <Text className="item-label">方案类型</Text>
                  <Text className="item-value">
                    {history1[0]?.plan || '-'}
                  </Text>
                </View>
                <View className="compare-item">
                  <Text className="item-label">方案类型</Text>
                  <Text className="item-value">
                    {history2[0]?.plan || '-'}
                  </Text>
                </View>
              </View>
            </View>
          )}

          {/* 康复要求完成情况对比 */}
          {history1.length > 0 && history2.length > 0 && (
            <View className="compare-section">
              <Text className="section-title">康复要求完成情况对比</Text>
              
              {[1, 2, 3, 4].map((num) => {
                const key = `requirement${num}Completed` as keyof RequirementStatus;
                const v1 = history1[0]?.requirements?.[key];
                const v2 = history2[0]?.requirements?.[key];

                return (
                  <View key={num} className="compare-row">
                    <View className="compare-item">
                      <Text className="item-label">要求{num}</Text>
                      <Text className={`item-value ${v1 ? 'text-green-500' : 'text-red-500'}`}>
                        {v1 ? '✓ 已完成' : '✗ 未完成'}
                      </Text>
                    </View>
                    <View className="compare-item">
                      <Text className="item-label">要求{num}</Text>
                      <Text className={`item-value ${v2 ? 'text-green-500' : 'text-red-500'}`}>
                        {v2 ? '✓ 已完成' : '✗ 未完成'}
                      </Text>
                    </View>
                  </View>
                );
              })}
            </View>
          )}

          {/* 历史记录 */}
          {history1.length > 1 || history2.length > 1 ? (
            <View className="compare-section">
              <Text className="section-title">历史记录对比</Text>
              
              {history1.length > 1 && (
                <View className="history-card">
                  <Text className="history-title">用户1历史记录</Text>
                  {history1.slice(1).map((record, index) => (
                    <View key={index} className="history-item">
                      <Text className="history-date">{record.date}</Text>
                      <Text className="history-detail">
                        BMI: {record.bmi?.toFixed(1)} | 整体健康: {record.healthAnalysis?.overallHealth}分
                      </Text>
                    </View>
                  ))}
                </View>
              )}

              {history2.length > 1 && (
                <View className="history-card">
                  <Text className="history-title">用户2历史记录</Text>
                  {history2.slice(1).map((record, index) => (
                    <View key={index} className="history-item">
                      <Text className="history-date">{record.date}</Text>
                      <Text className="history-detail">
                        BMI: {record.bmi?.toFixed(1)} | 整体健康: {record.healthAnalysis?.overallHealth}分
                      </Text>
                    </View>
                  ))}
                </View>
              )}
            </View>
          ) : null}
        </ScrollView>
      )}

      {/* 说明文字 */}
      {!showResult && (
        <View className="info-section">
          <Text className="info-text">
            说明：输入用户ID（如：user1, user2）进行搜索和对比
          </Text>
          <Text className="info-text">
            功能：支持基本信息、BMI、血压、健康要素、症状、方案、康复要求等多维度对比
          </Text>
        </View>
      )}
    </View>
  );
}
```
