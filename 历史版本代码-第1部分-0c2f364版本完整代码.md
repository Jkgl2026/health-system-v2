```markdown
# 中医深入分析功能集成文档

## 概述

本文档描述了如何将中医深入分析功能集成到后台用户详细信息页面中。

## 功能模块

### 1. 体质辨识
- 8种体质类型：平和质、气虚质、阳虚质、阴虚质、血瘀质、痰湿质、湿热质、气郁质
- 基于用户的健康数据自动辨识体质
- 提供体质描述和调理建议

### 2. 气血状态
- 4种气血状态：气血两虚、气虚血瘀、气血瘀滞、气血充盈
- 分析气血盈亏和运行情况
- 提供调理方案

### 3. 脏腑功能
- 5个脏腑评估：心、肝、脾、肺、肾
- 每个脏腑的功能评分（0-100）
- 脏腑功能描述和建议

### 4. 经络状态
- 4条经络分析：督脉、任脉、冲脉、带脉
- 经络通畅度评估
- 经络调理建议

### 5. 阴阳平衡
- 4种阴阳状态：阳盛阴衰、阴盛阳衰、阴阳两虚、阴阳平衡
- 阴阳平衡度评估
- 调理建议

### 6. 湿热寒凉
- 6种状态：寒湿、湿热、寒证、热证、湿证、燥证
- 体内环境分析
- 调理建议

## 集成位置

在健康要素分析结果之后，四个要求完成情况之前。

## UI设计

- 使用紫色渐变背景
- 浅色卡片设计（from-xxx-50 to-xxx-100）
- 确保文字可读性（text-gray-900, text-gray-600, text-gray-500）
- 优化间距和布局

## 数据来源

基于用户的以下数据：
- 症状检查数据
- 健康分析数据
- 健康方案数据
- 康复要求完成情况

## 实现步骤

1. 创建中医分析函数 `analyzeTCMHealth()`
2. 创建中医分析显示组件
3. 集成到用户详细信息页面
4. 测试和优化
```

## 3️⃣ src/app/admin/dashboard/page.tsx - 完整代码

```typescript
'use client';

import { useState, useEffect } from 'react';
import { 
  View, 
  Text, 
  Button, 
  Loading, 
  Input,
  ScrollView 
} from '@tarojs/components';
import Taro, { useRouter } from '@tarojs/taro';
import { adminLogout, isAdminLoggedIn } from '../../../utils/adminAuth';
import { getAllUsers, getUserRecords, getStatistics } from '../../../utils/cloud';
import { calculateComprehensiveHealthScore, getHealthLevel, getHealthSuggestion, calculateBMI, getBMILevel, getBMISuggestion } from '../../../utils/healthScore';
import './index.css';

interface UserInfo {
  _id: string;
  name: string;
  phone: string;
  gender: string;
  age: number;
  height: string;
  weight: string;
  occupation: string;
  created_at: string;
}

interface SymptomCheck {
  _id: string;
  symptoms: number[];
  totalScore: number;
  checkedAt: string;
}

interface HealthAnalysis {
  _id: string;
  qiAndBlood: number;
  circulation: number;
  toxins: number;
  bloodLipids: number;
  coldness: number;
  immunity: number;
  emotions: number;
  overallHealth: number;
}

interface HealthPlan {
  _id: string;
  planType: string;
  planDescription: string;
}

interface RequirementStatus {
  _id: string;
  requirement1Completed: boolean;
  requirement2Completed: boolean;
  requirement3Completed: boolean;
  requirement4Completed: boolean;
}

interface Statistics {
  totalUsers: number;
  totalChecks: number;
  totalAnalysis: number;
  totalPlans: number;
}

export default function AdminDashboard() {
  const [loading, setLoading] = useState(false);
  const [statistics, setStatistics] = useState<Statistics | null>(null);
  const [users, setUsers] = useState<UserInfo[]>([]);
  const [searchKeyword, setSearchKeyword] = useState('');
  const [selectedUser, setSelectedUser] = useState<UserInfo | null>(null);
  const [userSymptoms, setUserSymptoms] = useState<SymptomCheck[]>([]);
  const [userAnalysis, setUserAnalysis] = useState<HealthAnalysis[]>([]);
  const [userPlans, setUserPlans] = useState<HealthPlan[]>([]);
  const [userRequirements, setUserRequirements] = useState<RequirementStatus[]>([]);
  const [tcmAnalysis, setTcmAnalysis] = useState<any>(null);

  useEffect(() => {
    checkLoginStatus();
    loadStatistics();
    loadUsers();
  }, []);

  const checkLoginStatus = () => {
    if (!isAdminLoggedIn()) {
      Taro.redirectTo({
        url: '/pages/admin/login/index',
      });
    }
  };

  const loadStatistics = async () => {
    try {
      const res = await getStatistics();
      if (res.success) {
        setStatistics(res.data);
      }
    } catch (error) {
      console.error('[Dashboard] 加载统计数据失败:', error);
    }
  };

  const loadUsers = async () => {
    setLoading(true);
    try {
      const res = await getAllUsers({ limit: 50 });
      if (res.success) {
        setUsers(res.data.users || []);
      }
    } catch (error) {
      console.error('[Dashboard] 加载用户列表失败:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleSearch = () => {
    if (!searchKeyword) {
      loadUsers();
      return;
    }

    const filtered = users.filter(user => 
      user.name.includes(searchKeyword) || 
      user.phone.includes(searchKeyword)
    );
    setUsers(filtered);
  };

  const handleSelectUser = async (user: UserInfo) => {
    setSelectedUser(user);
    setLoading(true);

    try {
      // 加载用户的所有记录
      const recordsRes = await getUserRecords(user._id);
      if (recordsRes.success) {
        setUserSymptoms(recordsRes.data.symptoms || []);
        setUserAnalysis(recordsRes.data.analysis || []);
        setUserPlans(recordsRes.data.plans || []);
        setUserRequirements(recordsRes.data.requirements || []);

        // 如果有症状数据，生成中医分析
        if (recordsRes.data.symptoms && recordsRes.data.symptoms.length > 0) {
          const latestSymptom = recordsRes.data.symptoms[0];
          const tcmData = analyzeTCMHealth(user, latestSymptom);
          setTcmAnalysis(tcmData);
        }
      }
    } catch (error) {
      console.error('[Dashboard] 加载用户数据失败:', error);
    } finally {
      setLoading(false);
    }
  };

  // 中医深入分析函数
  const analyzeTCMHealth = (user: UserInfo, symptom: SymptomCheck) => {
    // 体质辨识
    const体质类型 = calculateConstitution(user, symptom);
    
    // 气血状态
    const气血状态 = calculateQiBlood(user, symptom);
    
    // 脏腑功能
    const脏腑功能 = calculateOrgans(user, symptom);
    
    // 经络状态
    const经络状态 = calculateMeridians(user, symptom);
    
    // 阴阳平衡
    const阴阳平衡 = calculateYinYang(user, symptom);
    
    // 湿热寒凉
    const湿热寒凉 = calculateColdHeat(user, symptom);

    return {
      体质类型,
      气血状态,
      脏腑功能,
      经络状态,
      阴阳平衡,
      湿热寒凉,
    };
  };

  // 体质辨识
  const calculateConstitution = (user: UserInfo, symptom: SymptomCheck) => {
    const symptoms = symptom.symptoms;
    let score: { [key: string]: number } = {
      平和质: 0,
      气虚质: 0,
      阳虚质: 0,
      阴虚质: 0,
      血瘀质: 0,
      痰湿质: 0,
      湿热质: 0,
      气郁质: 0,
    };

    // 根据症状计算各体质得分
    symptoms.forEach(id => {
      if (id >= 101 && id <= 103) {
        // 头部症状
        score.气虚质 += 2;
        score.阳虚质 += 1;
        score.阴虚质 += 2;
      } else if (id >= 201 && id <= 203) {
        // 身体症状
        score.气虚质 += 2;
        score.痰湿质 += 2;
        score.血瘀质 += 1;
      } else if (id >= 301 && id <= 303) {
        // 情绪症状
        score.气郁质 += 3;
        score.阴虚质 += 2;
        score.阳虚质 += 1;
      } else if (id >= 401 && id <= 403) {
        // 其他症状
        score.血瘀质 += 2;
        score.气虚质 += 1;
        score.阳虚质 += 2;
      }
    });

    // 根据用户信息调整
    if (user.age > 40) {
      score.阴虚质 += 1;
      score.气虚质 += 1;
    }
    if (user.gender === '女') {
      score.血瘀质 += 1;
      score.气郁质 += 1;
    }

    // 找出得分最高的体质
    const maxScore = Math.max(...Object.values(score));
    const constitution = Object.keys(score).find(key => score[key] === maxScore) || '平和质';

    return {
      type: constitution,
      description: getConstitutionDescription(constitution),
      score: maxScore,
    };
  };

  // 体质描述
  const getConstitutionDescription = (type: string): string => {
    const descriptions: { [key: string]: string } = {
      平和质: '阴阳气血调和，体态适中，面色红润，精力充沛，身体机能正常。',
      气虚质: '元气不足，气息低弱，容易疲劳，容易出汗，免疫力较低。',
      阳虚质: '阳气不足，畏寒怕冷，手足不温，精神不振，消化功能弱。',
      阴虚质: '阴液亏虚，口干咽燥，手足心热，心烦易怒，形体消瘦。',
      血瘀质: '血行不畅，面色晦暗，容易出现瘀斑，疼痛固定，舌质紫暗。',
      痰湿质: '体内痰湿偏重，形体肥胖，腹部肥满，胸闷痰多，口中黏腻。',
      湿热质: '湿热内蕴，面垢油光，容易长痘，口苦口臭，大便黏滞。',
      气郁质: '气机郁滞，精神抑郁，情绪不稳，胸闷不舒，叹息频作。',
    };
    return descriptions[type] || '';
  };

  // 气血状态计算
  const calculateQiBlood = (user: UserInfo, symptom: SymptomCheck) => {
    const score = symptom.totalScore;
    let state = '气血充盈';

    if (score >= 40) {
      state = '气血两虚';
    } else if (score >= 25) {
      state = '气血瘀滞';
    } else if (score >= 15) {
      state = '气虚血瘀';
    }

    return {
      state,
      description: getQiBloodDescription(state),
      level: score,
    };
  };

  // 气血状态描述
  const getQiBloodDescription = (state: string): string => {
    const descriptions: { [key: string]: string } = {
      气血两虚: '气血不足，面色苍白，容易头晕乏力，需要补气养血。',
      气虚血瘀: '气虚导致血行不畅，容易胸闷气短，需要益气活血。',
      气血瘀滞: '气血运行不畅，容易疼痛不适，需要行气活血。',
      气血充盈: '气血充足，面色红润，精力充沛，保持良好状态。',
    };
    return descriptions[state] || '';
  };

  // 脏腑功能计算
  const calculateOrgans = (user: UserInfo, symptom: SymptomCheck) => {
    const symptoms = symptom.symptoms;
    const organs = {
      心: 70,
      肝: 70,
      脾: 70,
      肺: 70,
      肾: 70,
    };

    // 根据症状调整脏腑功能
    symptoms.forEach(id => {
      if (id >= 101 && id <= 103) {
        // 头部症状影响心和肾
        organs.心 -= 5;
        organs.肾 -= 5;
      } else if (id >= 201 && id <= 203) {
        // 身体症状影响脾和胃
        organs.脾 -= 8;
      } else if (id >= 301 && id <= 303) {
        // 情绪症状影响肝
        organs.肝 -= 10;
      } else if (id >= 401 && id <= 403) {
        // 其他症状影响肺和肾
        organs.肺 -= 6;
        organs.肾 -= 6;
      }
    });

    // 根据用户信息调整
    if (user.age > 50) {
      organs.肾 -= 10;
    }

    return organs;
  };

  // 经络状态计算
  const calculateMeridians = (user: UserInfo, symptom: SymptomCheck) => {
    const score = symptom.totalScore;
    const meridians = {
      督脉: '通畅',
      任脉: '通畅',
      冲脉: '通畅',
      带脉: '通畅',
    };

    if (score >= 40) {
      meridians.督脉 = '欠通畅';
      meridians.任脉 = '欠通畅';
      meridians.冲脉 = '欠通畅';
      meridians.带脉 = '欠通畅';
    } else if (score >= 25) {
      meridians.督脉 = '欠通畅';
      meridians.任脉 = '欠通畅';
    }

    return {
      督脉: {
        state: meridians.督脉,
        description: meridians.督脉 === '通畅' ? '督脉运行正常，阳气充足' : '督脉欠通畅，需要疏通督脉',
      },
      任脉: {
        state: meridians.任脉,
        description: meridians.任脉 === '通畅' ? '任脉运行正常，阴血充足' : '任脉欠通畅，需要调理任脉',
      },
      冲脉: {
        state: meridians.冲脉,
        description: meridians.冲脉 === '通畅' ? '冲脉运行正常，气血调和' : '冲脉欠通畅，需要调理冲脉',
      },
      带脉: {
        state: meridians.带脉,
        description: meridians.带脉 === '通畅' ? '带脉运行正常，约束有力' : '带脉欠通畅，需要调理带脉',
      },
    };
  };

  // 阴阳平衡计算
  const calculateYinYang = (user: UserInfo, symptom: SymptomCheck) => {
    const symptoms = symptom.symptoms;
    let yinScore = 50;
    let yangScore = 50;

    symptoms.forEach(id => {
      if (id >= 301 && id <= 303) {
        // 情绪症状消耗阴液
        yinScore -= 5;
      } else if (id >= 401 && id <= 403) {
        // 其他症状消耗阳气
        yangScore -= 5;
      }
    });

    let state = '阴阳平衡';
    if (yinScore < yangScore - 10) {
      state = '阴盛阳衰';
    } else if (yangScore < yinScore - 10) {
      state = '阳盛阴衰';
    } else if (yinScore < 40 && yangScore < 40) {
      state = '阴阳两虚';
    }

    return {
      state,
      description: getYinYangDescription(state),
      yinScore,
      yangScore,
    };
  };

  // 阴阳平衡描述
  const getYinYangDescription = (state: string): string => {
    const descriptions: { [key: string]: string } = {
      阳盛阴衰: '阳气过盛，阴液不足，容易发热、烦躁，需要滋阴降火。',
      阴盛阳衰: '阴气过盛，阳气不足，容易畏寒、乏力，需要温阳散寒。',
      阴阳两虚: '阴液和阳气都不足，身体机能下降，需要阴阳双补。',
      阴阳平衡: '阴阳平衡协调，身体机能正常，继续保持。',
    };
    return descriptions[state] || '';
  };

  // 湿热寒凉计算
  const calculateColdHeat = (user: UserInfo, symptom: SymptomCheck) => {
    const symptoms = symptom.symptoms;
    let coldScore = 0;
    let heatScore = 0;
    let dampScore = 0;
    let dryScore = 0;

    symptoms.forEach(id => {
      if (id >= 101 && id <= 103) {
        // 头部症状
        heatScore += 2;
        dryScore += 1;
      } else if (id >= 201 && id <= 203) {
        // 身体症状
        coldScore += 2;
        dampScore += 2;
      } else if (id >= 301 && id <= 303) {
        // 情绪症状
        heatScore += 2;
      } else if (id >= 401 && id <= 403) {
        // 其他症状
        coldScore += 1;
        dampScore += 1;
      }
    });

    let state = '正常';
    if (dampScore >= 6) {
      state = '湿热';
    } else if (coldScore >= 6) {
      state = '寒湿';
    } else if (heatScore >= 6) {
      state = '热证';
    } else if (coldScore >= 4 && coldScore < 6) {
      state = '寒证';
    } else if (dampScore >= 4 && dampScore < 6) {
      state = '湿证';
    } else if (dryScore >= 4) {
      state = '燥证';
    }

    return {
      state,
      description: getColdHeatDescription(state),
      coldScore,
      heatScore,
      dampScore,
      dryScore,
    };
  };

  // 湿热寒凉描述
  const getColdHeatDescription = (state: string): string => {
    const descriptions: { [key: string]: string } = {
      寒湿: '寒湿内阻，容易腹泻、关节痛，需要温阳祛湿。',
      湿热: '湿热内蕴，容易长痘、口苦，需要清热祛湿。',
      寒证: '寒邪侵袭，容易畏寒、腹痛，需要温中散寒。',
      热证: '热邪内盛，容易发热、口干，需要清热解毒。',
      湿证: '湿邪内停，容易困倦、腹胀，需要健脾祛湿。',
      燥证: '燥邪伤津，容易口干、皮肤干，需要滋阴润燥。',
      正常: '体内环境平衡，继续保持。',
    };
    return descriptions[state] || '';
  };

  const handleLogout = () => {
    Taro.showModal({
      title: '提示',
      content: '确定要退出登录吗？',
      success: (res) => {
        if (res.confirm) {
          adminLogout();
          Taro.redirectTo({
            url: '/pages/admin/login/index',
          });
        }
      },
    });
  };

  const handleBack = () => {
    setSelectedUser(null);
    setUserSymptoms([]);
    setUserAnalysis([]);
    setUserPlans([]);
    setUserRequirements([]);
    setTcmAnalysis(null);
  };

  return (
    <View className="admin-dashboard">
      {!selectedUser ? (
        <View className="dashboard-view">
          {/* 头部 */}
          <View className="header">
            <Text className="header-title">数据概览</Text>
            <Button className="logout-btn" size="mini" onClick={handleLogout}>
              退出
            </Button>
          </View>

          {/* 统计卡片 */}
          <View className="stats-container">
            <View className="stat-card">
              <Text className="stat-value">{statistics?.totalUsers || 0}</Text>
              <Text className="stat-label">总用户数</Text>
            </View>
            <View className="stat-card">
              <Text className="stat-value">{statistics?.totalChecks || 0}</Text>
              <Text className="stat-label">完成自检</Text>
            </View>
            <View className="stat-card">
              <Text className="stat-value">{statistics?.totalAnalysis || 0}</Text>
              <Text className="stat-label">完成分析</Text>
            </View>
            <View className="stat-card">
              <Text className="stat-value">{statistics?.totalPlans || 0}</Text>
              <Text className="stat-label">已选方案</Text>
            </View>
          </View>

          {/* 搜索栏 */}
          <View className="search-bar">
            <Input
              className="search-input"
              placeholder="搜索用户姓名或手机号"
              value={searchKeyword}
              onInput={(e) => setSearchKeyword(e.detail.value)}
            />
            <Button className="search-btn" onClick={handleSearch}>
              搜索
            </Button>
          </View>

          {/* 用户列表 */}
          <ScrollView className="user-list" scrollY>
            {loading && <Loading />}
            {users.map((user, index) => (
              <View key={user._id || index} className="user-item" onClick={() => handleSelectUser(user)}>
                <Text className="user-name">{user.name || '未命名'}</Text>
                <Text className="user-phone">{user.phone || '暂无'}</Text>
                <Text className="user-age">{user.age}岁</Text>
              </View>
            ))}
          </ScrollView>
        </View>
      ) : (
        <View className="user-detail-view">
          {/* 用户信息 */}
          <View className="user-info-card">
            <View className="card-header">
```
