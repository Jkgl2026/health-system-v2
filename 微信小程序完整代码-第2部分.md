```typescript
import { useState, useEffect } from 'react';
import { View, Text, ScrollView, Checkbox, Button, Loading } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { SymptomCheck } from '../../types';
import { getUserRecords } from '../../utils/cloud';
import './index.css';

// 症状列表
const SYMPTOMS = [
  { id: 1, category: '头部', items: [
    { id: 101, name: '头痛', description: '经常性或间歇性头痛' },
    { id: 102, name: '头晕', description: '头晕目眩，站立不稳' },
    { id: 103, name: '失眠', description: '入睡困难或睡眠质量差' },
  ]},
  { id: 2, category: '身体', items: [
    { id: 201, name: '乏力', description: '全身无力，精神不振' },
    { id: 202, name: '食欲不振', description: '不想吃饭，消化不良' },
    { id: 203, name: '便秘', description: '排便困难，次数减少' },
  ]},
  { id: 3, category: '情绪', items: [
    { id: 301, name: '焦虑', description: '容易紧张，担心过度' },
    { id: 302, name: '抑郁', description: '情绪低落，缺乏兴趣' },
    { id: 303, name: '易怒', description: '情绪波动大，容易发脾气' },
  ]},
  { id: 4, category: '其他', items: [
    { id: 401, name: '心悸', description: '心慌，心跳加速' },
    { id: 402, name: '出汗', description: '盗汗或自汗' },
    { id: 403, name: '关节痛', description: '关节疼痛，活动受限' },
  ]},
];

export default function Check() {
  const [loading, setLoading] = useState(false);
  const [checkedSymptoms, setCheckedSymptoms] = useState<number[]>([]);
  const [userInfo, setUserInfo] = useState<any>(null);

  useEffect(() => {
    const app = getApp<IAppOption>();
    setUserInfo(app.globalData.userInfo);
  }, []);

  const handleCheckChange = (symptomId: number, checked: boolean) => {
    if (checked) {
      setCheckedSymptoms([...checkedSymptoms, symptomId]);
    } else {
      setCheckedSymptoms(checkedSymptoms.filter(id => id !== symptomId));
    }
  };

  const calculateScore = (): number => {
    // 每个症状5分
    return checkedSymptoms.length * 5;
  };

  const handleSubmit = async () => {
    if (checkedSymptoms.length === 0) {
      Taro.showToast({
        title: '请至少选择一项症状',
        icon: 'none',
      });
      return;
    }

    setLoading(true);

    try {
      const score = calculateScore();
      
      const symptomCheck: SymptomCheck = {
        userId: userInfo._id || '',
        userName: userInfo.nickName,
        phone: userInfo.phone,
        symptoms: checkedSymptoms,
        totalScore: score,
        checkedAt: new Date(),
        createTime: new Date(),
      };

      // 保存到云数据库
      await Taro.cloud.callFunction({
        name: 'user',
        data: {
          action: 'saveSymptomCheck',
          data: symptomCheck,
        },
      });

      // 跳转到结果页
      Taro.navigateTo({
        url: `/pages/result/index?score=${score}&count=${checkedSymptoms.length}`,
      });
    } catch (error) {
      console.error('[Check] 提交失败:', error);
      Taro.showToast({
        title: '提交失败，请重试',
        icon: 'none',
      });
    } finally {
      setLoading(false);
    }
  };

  return (
    <View className="check-page">
      {/* 头部 */}
      <View className="header">
        <Text className="header-title">症状自检</Text>
        <Text className="header-subtitle">请勾选您当前的症状</Text>
      </View>

      {/* 症状列表 */}
      <ScrollView className="symptom-list" scrollY>
        {SYMPTOMS.map(category => (
          <View key={category.id} className="category">
            <Text className="category-title">{category.category}</Text>
            {category.items.map(symptom => (
              <View key={symptom.id} className="symptom-item">
                <Checkbox
                  checked={checkedSymptoms.includes(symptom.id)}
                  onChange={(e) => handleCheckChange(symptom.id, e.detail.value)}
                >
                  <View className="symptom-content">
                    <Text className="symptom-name">{symptom.name}</Text>
                    <Text className="symptom-desc">{symptom.description}</Text>
                  </View>
                </Checkbox>
              </View>
            ))}
          </View>
        ))}
      </ScrollView>

      {/* 底部信息 */}
      <View className="footer-info">
        <Text className="selected-count">
          已选择 {checkedSymptoms.length} 项症状
        </Text>
        <Text className="total-score">
          预计得分: {calculateScore()} 分
        </Text>
      </View>

      {/* 提交按钮 */}
      <View className="footer">
        <Button 
          className="submit-btn"
          onClick={handleSubmit}
          disabled={loading || checkedSymptoms.length === 0}
        >
          {loading ? <Loading /> : '提交自检'}
        </Button>
      </View>
    </View>
  );
}
```

### pages/result/index.tsx - 自检结果页

```typescript
import { useState, useEffect } from 'react';
import { View, Text, Image, Button } from '@tarojs/components';
import Taro, { useRouter } from '@tarojs/taro';
import './index.css';

export default function Result() {
  const router = useRouter();
  const [score, setScore] = useState<number>(0);
  const [count, setCount] = useState<number>(0);

  useEffect(() => {
    const scoreStr = router.params.score || '0';
    const countStr = router.params.count || '0';
    setScore(parseInt(scoreStr));
    setCount(parseInt(countStr));
  }, []);

  const getLevel = (score: number): string => {
    if (score >= 40) return '较差';
    if (score >= 25) return '一般';
    if (score >= 10) return '良好';
    return '优秀';
  };

  const getAdvice = (score: number): string => {
    if (score >= 40) {
      return '您的健康状况需要重点关注，建议尽快进行健康分析，制定调理方案。';
    }
    if (score >= 25) {
      return '您的健康状况一般，建议进行健康分析，了解具体改善方向。';
    }
    if (score >= 10) {
      return '您的健康状况良好，但仍有一些需要注意的地方，建议继续进行健康分析。';
    }
    return '您的健康状况优秀，继续保持良好的生活习惯！';
  };

  const handleAnalysis = () => {
    Taro.navigateTo({
      url: '/pages/analysis/index',
    });
  };

  const handleBack = () => {
    Taro.navigateBack();
  };

  return (
    <View className="result-page">
      {/* 结果卡片 */}
      <View className="result-card">
        <Image 
          className="result-icon"
          src={`/assets/images/result-${getLevel(score)}.png`}
          mode="aspectFit"
        />
        <Text className="result-score">{score} 分</Text>
        <Text className="result-level">健康状况：{getLevel(score)}</Text>
      </View>

      {/* 统计信息 */}
      <View className="stats">
        <View className="stat-item">
          <Text className="stat-value">{count}</Text>
          <Text className="stat-label">症状数量</Text>
        </View>
        <View className="stat-divider" />
        <View className="stat-item">
          <Text className="stat-value">{score}</Text>
          <Text className="stat-label">总分</Text>
        </View>
      </View>

      {/* 建议 */}
      <View className="advice">
        <Text className="advice-title">💡 健康建议</Text>
        <Text className="advice-content">{getAdvice(score)}</Text>
      </View>

      {/* 操作按钮 */}
      <View className="actions">
        <Button 
          className="analysis-btn"
          onClick={handleAnalysis}
        >
          进行健康分析
        </Button>
        <Button 
          className="back-btn"
          onClick={handleBack}
        >
          返回自检
        </Button>
      </View>
    </View>
  );
}
```

### pages/analysis/index.tsx - 健康分析页

```typescript
import { useState, useEffect } from 'react';
import { View, Text, ScrollView, Button, Loading } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { HealthAnalysis } from '../../types';
import { calculateComprehensiveHealthScore, getHealthLevel, getHealthSuggestion } from '../../utils/healthScore';
import TrendChart from './components/TrendChart';
import './index.css';

const HEALTH_ELEMENTS = [
  { key: 'qiAndBlood', label: '气血', icon: '🔥', color: '#EF4444' },
  { key: 'circulation', label: '循环', icon: '❤️', color: '#3B82F6' },
  { key: 'toxins', label: '毒素', icon: '💧', color: '#F59E0B' },
  { key: 'bloodLipids', label: '血脂', icon: '🎯', color: '#EC4899' },
  { key: 'coldness', label: '寒凉', icon: '⚡', color: '#06B6D4' },
  { key: 'immunity', label: '免疫力', icon: '🛡️', color: '#10B981' },
  { key: 'emotions', label: '情绪', icon: '✨', color: '#8B5CF6' },
  { key: 'overallHealth', label: '整体健康', icon: '🏥', color: '#6366F1' },
];

export default function Analysis() {
  const [loading, setLoading] = useState(false);
  const [score, setScore] = useState<number>(0);
  const [healthAnalysis, setHealthAnalysis] = useState<HealthAnalysis | null>(null);

  useEffect(() => {
    // 从缓存获取上次自检得分
    const app = getApp<IAppOption>();
    setScore(app.globalData.lastScore || 0);
  }, []);

  const handleAnalyze = async () => {
    if (score === 0) {
      Taro.showToast({
        title: '请先进行症状自检',
        icon: 'none',
      });
      return;
    }

    setLoading(true);

    try {
      // 计算健康分析
      const analysis = calculateComprehensiveHealthScore(score, {});
      setHealthAnalysis(analysis);

      // 保存到云数据库
      const app = getApp<IAppOption>();
      await Taro.cloud.callFunction({
        name: 'user',
        data: {
          action: 'saveHealthAnalysis',
          data: {
            userId: app.globalData.userInfo._id,
            userName: app.globalData.userInfo.nickName,
            phone: app.globalData.userInfo.phone,
            ...analysis,
          },
        },
      });

      Taro.showToast({
        title: '分析完成',
        icon: 'success',
      });
    } catch (error) {
      console.error('[Analysis] 分析失败:', error);
      Taro.showToast({
        title: '分析失败，请重试',
        icon: 'none',
      });
    } finally {
      setLoading(false);
    }
  };

  return (
    <View className="analysis-page">
      {/* 头部 */}
      <View className="header">
        <Text className="header-title">健康分析</Text>
        <Text className="header-subtitle">基于您的自检结果生成健康报告</Text>
      </View>

      {/* 分析按钮 */}
      {!healthAnalysis && (
        <View className="analyze-section">
          <Text className="analyze-desc">
            您的上次自检得分为 {score} 分
          </Text>
          <Button 
            className="analyze-btn"
            onClick={handleAnalyze}
            disabled={loading}
          >
            {loading ? <Loading /> : '开始分析'}
          </Button>
        </View>
      )}

      {/* 健康分析结果 */}
      {healthAnalysis && (
        <ScrollView className="analysis-result" scrollY>
          {/* 整体健康评分 */}
          <View className="overall-health">
            <Text className="overall-title">整体健康评分</Text>
            <Text className="overall-score">{healthAnalysis.overallHealth}</Text>
            <Text className="overall-level">
              {getHealthLevel(healthAnalysis.overallHealth)}
            </Text>
          </View>

          {/* 健康要素 */}
          <View className="health-elements">
            <Text className="elements-title">健康要素分析</Text>
            {HEALTH_ELEMENTS.map(element => {
              const value = (healthAnalysis as any)[element.key];
              const level = getHealthLevel(value);
              const suggestion = getHealthSuggestion(element.key, value);

              return (
                <View key={element.key} className="element-item">
                  <View className="element-header">
                    <Text className="element-icon">{element.icon}</Text>
                    <Text className="element-label">{element.label}</Text>
                    <Text className={`element-score score-${level}`}>
                      {value}分
                    </Text>
                  </View>
                  <View className="element-level">{level}</View>
                  <Text className="element-suggestion">{suggestion}</Text>
                </View>
              );
            })}
          </View>

          {/* 趋势图表 */}
          <View className="chart-section">
            <Text className="chart-title">健康趋势</Text>
            <TrendChart data={healthAnalysis} />
          </View>

          {/* 建议 */}
          <View className="advice-section">
            <Text className="advice-title">💡 调理建议</Text>
            <Text className="advice-content">
              根据您的健康分析结果，建议您从以下方面进行调理：
            </Text>
            <Text className="advice-item">
              1. {getHealthSuggestion('overallHealth', healthAnalysis.overallHealth)}
            </Text>
            <Text className="advice-item">
              2. 定期进行症状自检，跟踪健康变化
            </Text>
            <Text className="advice-item">
              3. 保持良好的生活习惯，均衡饮食，适度运动
            </Text>
          </View>
        </ScrollView>
      )}
    </View>
  );
}
```

```typescript
import { useState, useEffect } from 'react';
import { View, Text, Image, Button } from '@tarojs/components';
import Taro from '@tarojs/taro';
import './index.css';

export default function My() {
  const [userInfo, setUserInfo] = useState<any>(null);

  useEffect(() => {
    const app = getApp<IAppOption>();
    setUserInfo(app.globalData.userInfo);
  }, []);

  const handleLogin = () => {
    Taro.getUserProfile({
      desc: '用于完善用户资料',
      success: (res) => {
        const app = getApp<IAppOption>();
        app.globalData.userInfo = res.userInfo;
        setUserInfo(res.userInfo);

            action: 'login',

    });

      url: '/pages/records/index',
```
